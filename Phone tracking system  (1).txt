    <script>
        // ============================================
        // COMPLETE PHONE TRACKER SYSTEM
        // Phone A generates code, shares location
        // Phone B enters code, sees live location on Google Maps
        // Uses localStorage + cross-tab sync
        // No external server needed - works entirely in browser
        // ============================================
        
        let map = null;
        let marker = null;
        let watchId = null;
        let isSharing = false;
        let currentCode = null;
        let trackingInterval = null;
        let isTracking = false;
        let currentTrackingCode = null;
        
        // Generate random 6-digit code
        function generateCode() {
            return Math.floor(100000 + Math.random() * 900000).toString();
        }
        
        // Initialize - check if code exists in localStorage
        function init() {
            // Try to load existing sharing session
            const savedCode = localStorage.getItem('phoneTracker_code');
            const savedSharing = localStorage.getItem('phoneTracker_sharing');
            
            if (savedCode && savedSharing === 'true') {
                currentCode = savedCode;
                document.getElementById('uniqueCode').innerText = currentCode;
                isSharing = true;
                document.getElementById('locationToggle').checked = true;
                startLocationSharing();
            } else {
                // Generate new code
                currentCode = generateCode();
                document.getElementById('uniqueCode').innerText = currentCode;
                localStorage.setItem('phoneTracker_code', currentCode);
            }
            
            // Check if we're currently tracking something
            const trackingCode = localStorage.getItem('phoneTracker_trackingCode');
            if (trackingCode) {
                document.getElementById('trackCodeInput').value = trackingCode;
                startTracking(trackingCode);
            }
            
            // Set up listeners for cross-tab sync
            window.addEventListener('storage', (e) => {
                if (e.key === 'phoneTracker_location_' + currentTrackingCode) {
                    // New location data available
                    if (isTracking && currentTrackingCode) {
                        const locationData = JSON.parse(e.newValue);
                        if (locationData && locationData.lat) {
                            updateMapLocation(locationData.lat, locationData.lng);
                            updateTrackerStatus(locationData);
                        }
                    }
                }
                if (e.key === 'phoneTracker_sharing') {
                    // Sharing status changed
                    if (e.newValue === 'false') {
                        if (isTracking) {
                            updateTrackerStatusText('Phone A stopped sharing location');
                            if (marker) marker.setMap(null);
                        }
                    }
                }
            });
        }
        
        // Start sharing location from Phone A
        function startLocationSharing() {
            if (!navigator.geolocation) {
                alert("Your browser doesn't support geolocation");
                return false;
            }
            
            // Update status
            document.getElementById('broadcasterDot').classList.add('active');
            document.getElementById('broadcasterStatusText').innerHTML = '📍 Sharing live location...';
            
            // Watch position continuously
            watchId = navigator.geolocation.watchPosition(
                (position) => {
                    const lat = position.coords.latitude;
                    const lng = position.coords.longitude;
                    const accuracy = position.coords.accuracy;
                    const timestamp = new Date().toLocaleTimeString();
                    
                    // Save location to localStorage with current code as key
                    const locationData = {
                        lat: lat,
                        lng: lng,
                        accuracy: accuracy,
                        timestamp: timestamp,
                        lastUpdate: Date.now()
                    };
                    
                    localStorage.setItem('phoneTracker_location_' + currentCode, JSON.stringify(locationData));
                    localStorage.setItem('phoneTracker_lastLocation', JSON.stringify(locationData));
                    
                    // Update UI
                    document.getElementById('broadcasterLocationInfo').innerHTML = 
                        `📍 Lat: ${lat.toFixed(6)}, Lng: ${lng.toFixed(6)}<br>🕐 Updated: ${timestamp} | Accuracy: ${Math.round(accuracy)}m`;
                    
                    // Also save that sharing is active
                    localStorage.setItem('phoneTracker_sharing', 'true');
                    localStorage.setItem('phoneTracker_code', currentCode);
                },
                (error) => {
                    console.error("Geolocation error:", error);
                    let errorMsg = "Location error: ";
                    switch(error.code) {
                        case 1: errorMsg += "Permission denied. Please allow location access."; break;
                        case 2: errorMsg += "Position unavailable."; break;
                        case 3: errorMsg += "Timeout. Check GPS signal."; break;
                        default: errorMsg += error.message;
                    }
                    document.getElementById('broadcasterStatusText').innerHTML = errorMsg;
                    document.getElementById('broadcasterDot').classList.remove('active');
                },
                {
                    enableHighAccuracy: true,
                    timeout: 10000,
                    maximumAge: 0
                }
            );
            
            return true;
        }
        
        // Stop sharing location
        function stopLocationSharing() {
            if (watchId !== null) {
                navigator.geolocation.clearWatch(watchId);
                watchId = null;
            }
            
            isSharing = false;
            document.getElementById('locationToggle').checked = false;
            document.getElementById('broadcasterDot').classList.remove('active');
            document.getElementById('broadcasterStatusText').innerHTML = '❌ Location sharing stopped';
            document.getElementById('broadcasterLocationInfo').innerHTML = '-';
            
            // Clear sharing flag
            localStorage.setItem('phoneTracker_sharing', 'false');
            // Remove location data
            if (currentCode) {
                localStorage.removeItem('phoneTracker_location_' + currentCode);
            }
        }
        
        // Initialize Google Map
        function initMap() {
            const defaultLocation = { lat: 28.6139, lng: 77.2090 }; // Default: Delhi
            map = new google.maps.Map(document.getElementById('map'), {
                zoom: 15,
                center: defaultLocation,
                mapTypeId: google.maps.MapTypeId.ROADMAP,
                styles: [
                    { elementType: "geometry", stylers: [{ color: "#242f3e" }] },
                    { elementType: "labels.text.stroke", stylers: [{ color: "#242f3e" }] },
                    { elementType: "labels.text.fill", stylers: [{ color: "#746855" }] },
                    {
                        featureType: "administrative.locality",
                        elementType: "labels.text.fill",
                        stylers: [{ color: "#d59563" }]
                    },
                    {
                        featureType: "poi",
                        elementType: "labels.text.fill",
                        stylers: [{ color: "#d59563" }]
                    },
                    {
                        featureType: "road",
                        elementType: "geometry",
                        stylers: [{ color: "#38414e" }]
                    }
                ]
            });
            
            marker = new google.maps.Marker({
                position: defaultLocation,
                map: map,
                icon: {
                    url: "http://maps.google.com/mapfiles/ms/icons/green-dot.png",
                    scaledSize: new google.maps.Size(40, 40)
                },
                title: "Target Phone"
            });
        }
        
        // Update map with new location
        function updateMapLocation(lat, lng) {
            if (!map || !marker) return;
            
            const position = new google.maps.LatLng(lat, lng);
            marker.setPosition(position);
            map.setCenter(position);
            
            // Update info
            document.getElementById('trackerLocationInfo').innerHTML = 
                `📍 Target location: ${lat.toFixed(6)}, ${lng.toFixed(6)}`;
        }
        
        // Update tracker status text
        function updateTrackerStatusText(text) {
            document.getElementById('trackerStatusText').innerHTML = text;
            if (text.includes("Tracking")) {
                document.getElementById('trackerDot').classList.add('active');
            } else {
                document.getElementById('trackerDot').classList.remove('active');
            }
        }
        
        function updateTrackerStatus(locationData) {
            if (locationData && locationData.lat) {
                updateTrackerStatusText(`🟢 Tracking live · Last update: ${locationData.timestamp || 'just now'}`);
                document.getElementById('trackerLocationInfo').innerHTML = 
                    `📍 Lat: ${locationData.lat.toFixed(6)}, Lng: ${locationData.lng.toFixed(6)}<br>📏 Accuracy: ${Math.round(locationData.accuracy)}m`;
            }
        }
        
        // Start tracking from Phone B
        function startTracking(code) {
            if (!code || code.length < 6) {
                alert("Please enter a valid 6-digit code");
                return false;
            }
            
            // Check if sharing is active on Phone A
            const sharingActive = localStorage.getItem('phoneTracker_sharing');
            const savedCode = localStorage.getItem('phoneTracker_code');
            
            if (savedCode !== code) {
                alert("❌ Invalid code! Make sure Phone A is sharing location with this code.");
                return false;
            }
            
            if (sharingActive !== 'true') {
                alert("⚠️ Phone A is not sharing location right now. Make sure location sharing is turned ON on Phone A.");
                return false;
            }
            
            // Clear any existing tracking
            if (trackingInterval) {
                clearInterval(trackingInterval);
            }
            
            isTracking = true;
            currentTrackingCode = code;
            localStorage.setItem('phoneTracker_trackingCode', code);
            
            // Show map
            document.getElementById('mapContainer').style.display = 'block';
            document.getElementById('stopTrackingBtn').style.display = 'block';
            
            // Initialize map if not already
            if (!map) {
                // Wait for map to initialize
                const checkMap = setInterval(() => {
                    if (typeof google !== 'undefined' && google.maps) {
                        clearInterval(checkMap);
                        initMap();
                        fetchAndDisplayLocation(code);
                    }
                }, 100);
            } else {
                fetchAndDisplayLocation(code);
            }
            
            // Set interval to fetch location every 3 seconds
            trackingInterval = setInterval(() => {
                if (isTracking && currentTrackingCode) {
                    fetchAndDisplayLocation(currentTrackingCode);
                }
            }, 3000);
            
            updateTrackerStatusText(`🟢 Tracking Phone A (${code})...`);
            return true;
        }
        
        function fetchAndDisplayLocation(code) {
            const locationDataStr = localStorage.getItem('phoneTracker_location_' + code);
            if (locationDataStr) {
                try {
                    const locationData = JSON.parse(locationDataStr);
                    const now = Date.now();
                    const lastUpdate = locationData.lastUpdate || 0;
                    
                    // Check if location is recent (within 10 seconds)
                    if (now - lastUpdate < 15000) {
                        updateMapLocation(locationData.lat, locationData.lng);
                        updateTrackerStatus(locationData);
                        updateTrackerStatusText(`🟢 Live tracking · Updated ${locationData.timestamp || 'now'}`);
                    } else {
                        updateTrackerStatusText(`⚠️ Waiting for location update from Phone A...`);
                        document.getElementById('trackerLocationInfo').innerHTML = '⏳ No recent location data. Make sure Phone A has GPS on.';
                    }
                } catch(e) {
                    console.error("Error parsing location:", e);
                }
            } else {
                updateTrackerStatusText(`⏳ Waiting for Phone A (${code}) to share location...`);
                document.getElementById('trackerLocationInfo').innerHTML = 'Make sure Phone A has turned ON location sharing.';
            }
        }
        
        function stopTracking() {
            if (trackingInterval) {
                clearInterval(trackingInterval);
                trackingInterval = null;
            }
            
            isTracking = false;
            currentTrackingCode = null;
            localStorage.removeItem('phoneTracker_trackingCode');
            
            document.getElementById('mapContainer').style.display = 'none';
            document.getElementById('stopTrackingBtn').style.display = 'none';
            updateTrackerStatusText('Not tracking. Enter code to start.');
            document.getElementById('trackerLocationInfo').innerHTML = '';
            document.getElementById('trackCodeInput').value = '';
        }
        
        // Event Listeners
        document.getElementById('copyCodeBtn').addEventListener('click', () => {
            const code = document.getElementById('uniqueCode').innerText;
            navigator.clipboard.writeText(code).then(() => {
                alert('Code copied: ' + code);
            }).catch(() => {
                prompt('Copy this code manually:', code);
            });
        });
        
        document.getElementById('locationToggle').addEventListener('change', (e) => {
            if (e.target.checked) {
                // Start sharing
                if (navigator.geolocation) {
                    startLocationSharing();
                } else {
                    alert("Geolocation not supported");
                    e.target.checked = false;
                }
            } else {
                // Stop sharing
                stopLocationSharing();
            }
        });
        
        document.getElementById('startTrackingBtn').addEventListener('click', () => {
            const code = document.getElementById('trackCodeInput').value.trim();
            if (code) {
                startTracking(code);
            } else {
                alert("Please enter the 6-digit code from Phone A");
            }
        });
        
        document.getElementById('stopTrackingBtn').addEventListener('click', () => {
            stopTracking();
        });
        
        document.getElementById('stopSharingBtn').addEventListener('click', () => {
            if (isSharing || watchId !== null) {
                stopLocationSharing();
            } else {
                alert("Location sharing is not active");
            }
        });
        
        // Restrict code input to numbers only
        document.getElementById('trackCodeInput').addEventListener('input', (e) => {
            e.target.value = e.target.value.replace(/[^0-9]/g, '').slice(0, 6);
        });
        
        // Check if browser supports geolocation on load
        if (!navigator.geolocation) {
            document.getElementById('broadcasterStatusText').innerHTML = '❌ Geolocation not supported';
            document.getElementById('locationToggle').disabled = true;
        }
        
        // Check for existing active tracking on load
        window.initMap = initMap;
        
        // Start automatically if previously sharing
        const wasSharing = localStorage.getItem('phoneTracker_sharing');
        if (wasSharing === 'true') {
            const savedCode = localStorage.getItem('phoneTracker_code');
            if (savedCode && navigator.geolocation) {
                document.getElementById('locationToggle').checked = true;
                startLocationSharing();
            }
        }
        
        // Call init
        init();
        
        // Handle page unload - don't automatically stop sharing
        window.addEventListener('beforeunload', () => {
            // We don't stop sharing on page unload so it persists
        });
    </script>
